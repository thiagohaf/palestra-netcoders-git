First Commands

  • Leia o manual do git para os comandos mais usados
  • Leia o feedback do git

git help <comand>

							
	$ git help
	$ git help init

git init

							
	$ git init

git clone <repository>

							
	$ git clone https://github.com/twbs/bootstrap.git

git status

							
	$ git status

git diff

							
	$ git diff
	$ git diff --staged

git add <filter>

							
	$ git add .

git commit <options>

							
	$ git commit -m "First commit"

Undo

git commit --amend

							
	$ git commit --amend -m "Nova mensagem do commit"

git reset HEAD <file>

							
	$ git reset HEAD index.html

git checkout -- <file>

							
	$ git checkout index.html

Remotes

git remote

							
	$ git remote
	$ git remote -v

git remote add [name] [url]

							
	$ git remote add github https://github.com/twbs/bootstrap.git

git pull [name]

							
	$ git pull
	$ git pull origin

git push [name] [branch]

							
	$ git push
	$ git push origin master
